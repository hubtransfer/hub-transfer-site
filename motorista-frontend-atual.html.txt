function calcFlightProgress(depT,arrT){
  if(!depT||!arrT)return 50;const now=new Date(),hN=now.getHours()*60+now.getMinutes();
  const p=(t)=>{const m=String(t).match(/(\d{1,2}):(\d{2})/);return m?parseInt(m[1])*60+parseInt(m[2]):null;};
  const d=p(depT),a=p(arrT);if(d===null||a===null)return 50;
  const dur=a>d?a-d:(a+1440)-d;if(dur<=0)return 50;
  const el=hN>=d?hN-d:(hN+1440)-d;if(el<0)return 5;if(el>=dur)return 95;
  return Math.round((el/dur)*90)+5;
}

function showNameplate(name){
  let ov=document.getElementById('nameplate');
  if(!ov){
    ov=document.createElement('div');ov.id='nameplate';ov.className='nameplate';
    ov.innerHTML=`<div class="nameplate-logo">HUB TRANSFER</div><div class="nameplate-name" id="npName"></div><div class="nameplate-hint">toca para fechar</div>`;
    ov.onclick=()=>{ov.classList.remove('open');document.body.style.overflow='';try{screen.orientation.unlock();}catch(e){}};
    document.body.appendChild(ov);
  }
  const el=document.getElementById('npName');
  el.textContent=name;
  ov.classList.add('open');
  document.body.style.overflow='hidden';
  /* Tentar forçar landscape */
  try{screen.orientation.lock('landscape').catch(()=>{});}catch(e){}
  /* Calcular tamanho máximo */
  requestAnimationFrame(()=>{
    let sz=20;el.style.fontSize=sz+'vw';
    const maxW=window.innerWidth*0.94,maxH=window.innerHeight*0.6;
    while((el.scrollWidth>maxW||el.scrollHeight>maxH)&&sz>3){sz-=0.5;el.style.fontSize=sz+'vw';}
  });
}
