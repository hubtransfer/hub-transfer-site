function guessDepAirport(flight){
  if(!flight)return null;
  const code=String(flight).replace(/\s/g,'');
  /* Tentar 2 letras primeiro, depois 3 (Ryanair FR, TAP TP, TUI BY, etc) */
  const m2=code.match(/^([A-Z]{2})\d/);
  if(m2 && AIRLINE_HUB[m2[1]]) return AIRLINE_HUB[m2[1]];
  const m3=code.match(/^([A-Z]{3})\d/);
  if(m3 && AIRLINE_HUB[m3[1]]) return AIRLINE_HUB[m3[1]];
  return null;
}

function calcFlightProgress(depTime, arrTime){
  /* Estima progresso do voo: 0-100 baseado na hora actual vs dep/arr */
  if(!depTime||!arrTime)return 50; /* fallback centro */
  const now = new Date();
  const hNow = now.getHours()*60 + now.getMinutes();
  const parseTm = (t)=>{const m=String(t).match(/(\d{1,2}):(\d{2})/); return m?parseInt(m[1])*60+parseInt(m[2]):null;};
  const dM=parseTm(depTime), aM=parseTm(arrTime);
  if(dM===null||aM===null)return 50;
  const dur = aM > dM ? aM-dM : (aM+1440)-dM; /* handle overnight */
  if(dur<=0)return 50;
  const elapsed = hNow >= dM ? hNow-dM : (hNow+1440)-dM;
  if(elapsed<0)return 5; /* ainda não partiu */
  if(elapsed>=dur)return 95; /* já aterrou */
  return Math.round((elapsed/dur)*90)+5;
}

function markDone(id){
  const i=SERVICES.current.findIndex(s=>s.id===id);
  if(i===-1)return;
  const s=SERVICES.current.splice(i,1)[0];
  s.doneAt=new Date().toLocaleString('pt-PT',{timeZone:'Europe/Lisbon',hour:'2-digit',minute:'2-digit'});
  SERVICES.past.unshift(s);
  renderAll();
}
function markCancelled(id){
  if(!confirm('Cancelar esta viagem?'))return;
  const i=SERVICES.current.findIndex(s=>s.id===id);
  if(i===-1)return;
  SERVICES.cancelled.unshift(SERVICES.current.splice(i,1)[0]);
  renderAll();
}
function restoreService(id,from){
  const arr=SERVICES[from];
  const i=arr.findIndex(s=>s.id===id);
  if(i===-1)return;
  SERVICES.current.push(arr.splice(i,1)[0]);
  switchTab('current',document.getElementById('sb-current'));
}
function removeService(id,from){
  if(!confirm('Eliminar permanentemente?'))return;
  SERVICES[from]=SERVICES[from].filter(s=>s.id!==id);
  renderAll();
}

/* ─── SEPARAR NOME DO ESTABELECIMENTO DO ENDEREÇO ─── */
function splitLocation(raw){
  if(!raw || raw==='—') return {name:'',addr:raw||'—',full:raw||'—'};
  const s = String(raw).trim();
  const parts = s.split(',').map(p=>p.trim());
  if(parts.length < 2) return {name:'',addr:s,full:s};
  const first = parts[0].toLowerCase();
  /* Detectar se o primeiro segmento é um nome de estabelecimento */
  const keywords = ['hotel','hostel','residencial','aparthotel','guest house','guesthouse',
    'pension','pousada','albergue','resort','villa','palace','palácio','suites','lodge',
    'inn','b&b','bed and','airbnb','flat','apartment','apartamento','residence','residência',
    'aeroporto','airport','terminal','humberto delgado','portela',
    'epic sana','sana','turim','ibis','novotel','mercure','marriott','hilton','sheraton',
    'radisson','holiday inn','best western','accor','melia','meliã','pestana','altis',
    'tivoli','vip','jupiter','dom pedro','corinthia','intercontinental','four seasons',
    'ritz','sofitel','crowne','hyatt','lux','inspira','memmo','bairro alto','the vintage',
    'lisbon','lisboa','heritage','rossio','marquês','marques','empire','gota',
    'escola','museu','museum','estação','station','praça','plaza','centro','center',
    'parque','park','jardim','garden','castelo','castle','palacio','miradouro',
    'restaurante','restaurant','café','cafe','bar','clube','club','igreja','church'];
  const hasKeyword = keywords.some(k => first.includes(k));
  /* Se começa com número (ex: "Rua X 123") provavelmente é endereço directo */
  const startsWithNumber = /^\d/.test(parts[0]);
  /* Se tem "Rua","Av.","R.","Alameda","Travessa","Largo","Praça" no início é endereço */
  const streetPrefixes = /^(rua|r\.|av\.|av |avenida|alameda|travessa|tv\.|largo|praça|pc\.|estrada|calçada|beco|campo|cais)/i;
  const isStreet = streetPrefixes.test(parts[0]);
  if(hasKeyword && !startsWithNumber && !isStreet){
    const nome = parts[0];
    const endereco = parts.slice(1).join(', ');
    return {name:nome, addr:endereco, full:s};
  }
  return {name:'', addr:s, full:s};
}

function cleanHora(h){
  if(!h) return '—:—';
  const s = String(h);
  /* Tentar extrair HH:MM de qualquer formato (Date obj, string, etc) */
  const m = s.match(/(\d{1,2}):(\d{2})/);
  if(m) return m[1].padStart(2,'0') + ':' + m[2];
  /* Se veio como Date object stringified, tentar parsear */
  try{
    const d = new Date(s);
    if(!isNaN(d.getTime())){
      return d.getHours().toString().padStart(2,'0') + ':' + d.getMinutes().toString().padStart(2,'0');
    }
  }catch(e){}
  return s.slice(0,5) || '—:—';
}

function detectLangFromPhone(phone){
  if(!phone) return null;
  const p = String(phone).replace(/\D/g,'');
  /* Tentar 3 dígitos, depois 2, depois 1 */
  if(DDI_LANG[p.slice(0,3)]) return DDI_LANG[p.slice(0,3)];
  if(DDI_LANG[p.slice(0,2)]) return DDI_LANG[p.slice(0,2)];
  if(DDI_LANG[p.slice(0,1)]) return DDI_LANG[p.slice(0,1)];
  return null;
}
function resolveLanguage(lang, phone){
  /* 1. Se o GAS enviou idioma válido (PT, EN, FR, ES, IT, DE), usar */
  const valid = ['PT','EN','ES','FR','IT','DE'];
  const upper = String(lang||'').toUpperCase().trim();
  if(upper && valid.includes(upper)) return upper;
  /* 2. Se só veio as 2 primeiras letras de algo mais longo, normalizar */
  if(upper.length >= 2 && valid.includes(upper.slice(0,2))) return upper.slice(0,2);
  /* 3. Tentar detectar pelo DDI do telefone */
  const fromPhone = detectLangFromPhone(phone);
  if(fromPhone && valid.includes(fromPhone)) return fromPhone;
  /* 4. Fallback EN */
  return 'EN';
}

/* ─── DAR BAIXA — Marcar viagem como concluída ─── */
async function darBaixa(id, rowIndex, cardId){
  if(!confirm('Confirmar conclusão desta viagem?')) return;
  const btn = document.getElementById('done-'+cardId);
  if(btn){ btn.classList.add('loading'); btn.textContent='⏳...'; }
  try{
    const url = HUB_VIAGENS_URL + '?action=completar&id=' + encodeURIComponent(id) + '&row=' + (rowIndex||'') + '&t=' + Date.now();
    const resp = await fetch(url, {redirect:'follow'});
    const data = await resp.json();
    if(data.success){
      if(btn){ btn.textContent='✅ Concluída'; btn.style.pointerEvents='none'; }
      const card = document.getElementById('dcard-'+cardId);
      if(card) card.classList.add('dc-done');
      /* Actualizar localmente */
      const v = HUB_VIAGENS.find(x=>(x.id||'')==id);
      if(v){ v.concluida = true; v.status = 'CONCLUIDA'; }
    } else {
      alert('⚠️ Erro: ' + (data.error||'Não foi possível dar baixa'));
      if(btn){ btn.classList.remove('loading'); btn.textContent='✅ Dar Baixa'; }
    }
  }catch(err){
    alert('⚠️ Erro de rede: ' + err.message);
    if(btn){ btn.classList.remove('loading'); btn.textContent='✅ Dar Baixa'; }
  }
}

async function syncViagens(manual){
  if(!HUB_VIAGENS_URL){ setViagensStatus('offline','URL não configurado'); if(currentTab==='dia')renderDia(); return; }
  setViagensStatus('loading','A sincronizar...');
  try{
    let url = HUB_VIAGENS_URL + '?action=viagens&t=' + Date.now();
    if(SELECTED_DATE) url += '&data=' + encodeURIComponent(SELECTED_DATE);
    const resp = await fetch(url, {redirect:'follow'});
    if(!resp.ok) throw new Error('HTTP ' + resp.status);
    const data = await resp.json();
    if(data.error) throw new Error(data.error);
    const list = Array.isArray(data) ? data : (data.viagens || []);
    HUB_VIAGENS = list;
    const ts = new Date().toLocaleString('pt-PT',{timeZone:'Europe/Lisbon',hour:'2-digit',minute:'2-digit'});
    localStorage.setItem('hub_viagens_last_sync', ts);
    const dateLabel = SELECTED_DATE || todayStr();
    setViagensStatus('online', `${list.length} viagens · ${dateLabel} · ${ts}`);
    updateCounts();
    if(manual){
      const btn = document.getElementById('sbViagensSync');
      if(btn){ const prev=btn.innerHTML; btn.innerHTML='<span>✅</span> OK!'; setTimeout(()=>btn.innerHTML=prev,2000); }
    }
  }catch(err){
    console.warn('Viagens sync failed:', err.message);
    setViagensStatus('offline','Erro: '+err.message.slice(0,30));
  }
  if(currentTab==='dia') renderDia();
}

async function syncDrivers(manual){
  setHubStatus('loading','A sincronizar...');
  try{
    const url=HUB_CENTRAL_URL+'?action=motoristas&t='+Date.now();
    const resp=await fetch(url,{redirect:'follow'});
    if(!resp.ok) throw new Error('HTTP '+resp.status);
    const data=await resp.json();
    const list=Array.isArray(data)?data:(data.motoristas||data.drivers||[]);
    if(!list.length && data.error) throw new Error(data.error);
    if(list.length===0){
      DRIVERS=DRIVERS_FALLBACK;
      setHubStatus('offline','Sem motoristas na aba',`<span>Aba vazia — a usar lista local</span>`);
    } else {
      DRIVERS=list.map(d=>({
        name:   String(d.name||d.nome||d.NOME||d.motorista||'').trim(),
        phone:  String(d.phone||d.telefone||d.tel||d.TELEFONE||'').replace(/[\s\-\+\(\)]/g,''),
        viatura:String(d.viatura||d.car||d.vehicle||'').trim(),
      })).filter(d=>d.name);
      const ts=new Date().toLocaleString('pt-PT',{timeZone:'Europe/Lisbon',hour:'2-digit',minute:'2-digit'});
      setHubStatus('online','HUB Central · Ligada',
        `<span class="hi-count">✦ ${DRIVERS.length} motoristas</span>`+
        `<span>Sync: ${ts}</span>`+
        DRIVERS.map(d=>`<span style="color:var(--text-dim)">· ${d.name}${d.viatura?' ('+d.viatura+')':''}</span>`).join('')
      );
      if(manual){
        const btn=document.getElementById('sbSyncBtn');
        const prev=btn.innerHTML;
        btn.innerHTML='<span>✅</span> Actualizado!';
        setTimeout(()=>btn.innerHTML=prev,2000);
      }
    }
  }catch(err){
    console.warn('HUB Central sync failed:',err.message);
    DRIVERS=DRIVERS_FALLBACK;
    setHubStatus('offline','Offline — lista local',
      `<span style="color:var(--danger);font-size:.48rem">${err.message}</span>`+
      `<span>A usar ${DRIVERS_FALLBACK.length} motoristas locais</span>`
    );
  }
  renderAll();
}

    }
  });
